# Testar normalidade das distribui√ß√µes
print("\nüî¨ TESTES DE NORMALIDADE (Shapiro-Wilk):")
print("-" * 50)

numerical_vars = ['customer_age', 'monthly_income', 'purchase_amount', 'delivery_days']

for var in numerical_vars:
    # Amostra menor para o teste (Shapiro-Wilk tem limite)
    sample = df[var].sample(1000, random_state=42) if len(df) > 1000 else df[var]
    
    statistic, p_value = stats.shapiro(sample)
    
    print(f"\nüìä {var.upper().replace('_', ' ')}:")
    print(f"   ‚Ä¢ Estat√≠stica: {statistic:.6f}")
    print(f"   ‚Ä¢ P-valor: {p_value:.6f}")
    
    if p_value > 0.05:
        print(f"   ‚Ä¢ ‚úÖ Distribui√ß√£o NORMAL (p > 0.05)")
        print(f"   ‚Ä¢ ü§ñ IA: Pode usar algoritmos que assumem normalidade")
    else:
        print(f"   ‚Ä¢ ‚ùå Distribui√ß√£o N√ÉO-NORMAL (p ‚â§ 0.05)")
        print(f"   ‚Ä¢ ü§ñ IA: Considerar transforma√ß√µes ou algoritmos n√£o-param√©tricos")
    
    # Calcular assimetria e curtose
    skewness = stats.skew(df[var])
    kurtosis = stats.kurtosis(df[var])
    
    print(f"   ‚Ä¢ Assimetria: {skewness:.3f} ", end="")
    if abs(skewness) < 0.5:
        print("(Aproximadamente sim√©trica)")
    elif skewness > 0:
        print("(Assim√©trica √† direita)")
    else:
        print("(Assim√©trica √† esquerda)")
    
    print(f"   ‚Ä¢ Curtose: {kurtosis:.3f} ", end="")
    if abs(kurtosis) < 0.5:
        print("(Normal)")
    elif kurtosis > 0:
        print("(Leptoc√∫rtica - mais pontiaguda)")
    else:
        print("(Platic√∫rtica - mais achatada)")
üîç An√°lise de Distribui√ß√µes:

Idade: Pr√≥xima √† normal (gerada propositalmente) | Renda: Log-normal (realista) - m√©dia > mediana | Compras: Influenciada pela renda | Avalia√ß√µes: Concentrada em valores altos | Entrega: Exponencial (maioria r√°pida, alguns demoram) | Os testes de normalidade orientam escolha de algoritmos de IA.

# Fun√ß√£o para detectar outliers pelo m√©todo IQR
def detect_outliers_iqr(data, column):
    Q1 = data[column].quantile(0.25)
    Q3 = data[column].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = data[(data[column] < lower_bound) | (data[column] > upper_bound)]
    
    return outliers, lower_bound, upper_bound, Q1, Q3, IQR

# Fun√ß√£o para detectar outliers pelo m√©todo Z-Score
def detect_outliers_zscore(data, column, threshold=3):
    z_scores = np.abs(stats.zscore(data[column]))
    outliers = data[z_scores > threshold]
    return outliers, z_scores
