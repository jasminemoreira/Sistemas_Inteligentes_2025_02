Caso Prático: Otimização de Estoque E-commerce

Vamos simular um problema real de uma empresa de e-commerce que precisa otimizar seu estoque para reduzir custos e evitar rupturas. Você aplicará estatística descritiva, análise de distribuições e correlações para gerar insights acionáveis.

🏪 Contexto do Problema

Empresa: TechStore Brasil - E-commerce de eletrônicos

Desafio: Alto custo de estoque parado (35% dos produtos) vs. frequentes rupturas (20% das vendas perdidas)

Objetivo: Identificar padrões de demanda para otimizar reposição de estoque

Dados: 6 meses de vendas (Jan-Jun 2024) com 15.000 transações

Meta: Reduzir estoque parado em 40% e ruptura em 50%

🏪 Setup do Caso TechStore 
# -*- coding: utf-8 -*-
"""
# Sistemas Inteligentes - Aula 3: Aplicações Práticas
## Caso: TechStore Brasil - Otimização de Estoque

**Problema de Negócio:** Como reduzir estoque parado e rupturas usando dados?
**Técnicas:** Estatística descritiva, análise de sazonalidade, correlações
**Objetivo:** Gerar insights para algoritmos de IA de previsão de demanda
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime, timedelta
from scipy import stats
import warnings
warnings.filterwarnings('ignore')

# Configurações
plt.style.use('seaborn-v0_8-darkgrid')
sns.set_palette("Set2")
plt.rcParams['figure.figsize'] = (14, 8)
np.random.seed(42)

print("🏪 TECHSTORE BRASIL - ANÁLISE DE ESTOQUE")
print("=" * 50)
print("📊 Ambiente configurado!")
print("🎯 Objetivo: Otimizar estoque usando estatística")
📊 Geração dos Dados da TechStore 
# Gerar dataset realista de vendas de eletrônicos
print("🔄 Gerando dados de vendas TechStore (Jan-Jun 2024)...")

n_vendas = 15000
# Produtos da TechStore
produtos = {
    'Smartphone': {'preco_medio': 800, 'sazonalidade': [1.2, 0.9, 1.1, 1.0, 0.8, 1.3]},
    'Notebook': {'preco_medio': 2500, 'sazonalidade': [0.8, 1.4, 1.2, 0.9, 0.7, 1.1]},
    'Tablet': {'preco_medio': 600, 'sazonalidade': [1.0, 1.1, 0.9, 1.2, 1.0, 0.8]},
    'Smartwatch': {'preco_medio': 400, 'sazonalidade': [1.1, 0.8, 1.3, 1.0, 1.2, 0.9]},
    'Fone Bluetooth': {'preco_medio': 150, 'sazonalidade': [0.9, 1.0, 1.1, 1.3, 1.2, 1.0]}
}

# Gerar datas com padrões realistas
start_date = datetime(2024, 1, 1)
end_date = datetime(2024, 6, 30)
dates = pd.date_range(start_date, end_date, freq='D')

vendas_data = []
product_id = 1

for date in dates:
    # Mais vendas em fins de semana e menos vendas em segundas
    day_multiplier = 1.3 if date.weekday() >= 5 else 0.7 if date.weekday() == 0 else 1.0
    
    # Vendas por produto no dia
    for produto, info in produtos.items():
        month_idx = date.month - 1
        seasonal_factor = info['sazonalidade'][month_idx]
        
        # Número de vendas do produto no dia
        base_vendas = np.random.poisson(4)  # Média 4 vendas/dia por produto
        daily_sales = int(base_vendas * day_multiplier * seasonal_factor)
        
        for _ in range(daily_sales):
            if len(vendas_data) >= n_vendas:
                break
                
            # Gerar venda individual
            preco_base = info['preco_medio']
            preco_final = preco_base + np.random.normal(0, preco_base * 0.1)  # Variação ±10%
            
            venda = {
                'product_id': f'PROD_{product_id:05d}',
                'product_name': produto,
                'sale_date': date,
                'sale_price': max(50, preco_final),  # Preço mínimo R$ 50
                'quantity': np.random.choice([1, 2, 3], p=[0.7, 0.25, 0.05]),
                'customer_segment': np.random.choice(['Premium', 'Standard', 'Budget'], p=[0.2, 0.5, 0.3]),
                'sales_channel': np.random.choice(['Online', 'App', 'Loja Física'], p=[0.5, 0.3, 0.2]),
                'region': np.random.choice(['Sudeste', 'Sul', 'Nordeste', 'Norte', 'Centro-Oeste'], 
                                         p=[0.45, 0.25, 0.15, 0.08, 0.07])
            }
            
            # Calcular revenue
            venda['revenue'] = venda['sale_price'] * venda['quantity']
            
            vendas_data.append(venda)
            product_id += 1
    
    if len(vendas_data) >= n_vendas:
        break

# Criar DataFrame
df_vendas = pd.DataFrame(vendas_data[:n_vendas])

# Adicionar variáveis derivadas
df_vendas['month'] = df_vendas['sale_date'].dt.month
df_vendas['weekday'] = df_vendas['sale_date'].dt.weekday
df_vendas['week_number'] = df_vendas['sale_date'].dt.isocalendar().week

print(f"✅ Dataset TechStore criado!")
print(f"📊 {len(df_vendas):,} vendas registradas")
print(f"📅 Período: {df_vendas['sale_date'].min().strftime('%d/%m/%Y')} a {df_vendas['sale_date'].max().strftime('%d/%m/%Y')}")
print(f"💰 Receita total: R$ {df_vendas['revenue'].sum():,.2f}")
print(f"\n📋 Primeiras 5 vendas:")
print(df_vendas.head())

Análise Exploratória Direcionada ao Negócio

Agora vamos aplicar estatística descritiva com foco no problema de negócio. Cada análise será direcionada para gerar insights sobre padrões de demanda, sazonalidade e comportamento de estoque.

📊 Análise 1: Padrões de Demanda por Produto 
print("📊 ANÁLISE 1: PADRÕES DE DEMANDA")
print("=" * 40)

# 1. Análise por produto
demanda_produto = df_vendas.groupby('product_name').agg({
    'quantity': ['sum', 'mean', 'std', 'count'],
    'revenue': ['sum', 'mean'],
    'sale_price': ['mean', 'std']
}).round(2)

demanda_produto.columns = ['_'.join(col).strip() for col in demanda_produto.columns]

print("📋 ESTATÍSTICAS POR PRODUTO:")
print(demanda_produto)

# Calcular coeficiente de variação da demanda (importante para estoque)
print(f"\n🎯 VARIABILIDADE DA DEMANDA (Coeficiente de Variação):")
for produto in df_vendas['product_name'].unique():
    produto_data = df_vendas[df_vendas['product_name'] == produto]['quantity']
    cv = (produto_data.std() / produto_data.mean()) * 100
    
    print(f"   {produto}: {cv:.1f}% ", end="")
    if cv > 50:
        print("🔴 ALTA (Estoque de segurança maior)")
    elif cv > 30:
        print("🟡 MODERADA (Monitorar padrões)")
    else:
        print("🟢 BAIXA (Demanda previsível)")

# Visualização da demanda
fig, axes = plt.subplots(2, 2, figsize=(16, 10))
fig.suptitle('TechStore - Análise de Demanda por Produto', fontsize=16, fontweight='bold')

# Quantidade vendida por produto
vendas_por_produto = df_vendas.groupby('product_name')['quantity'].sum().sort_values(ascending=False)
axes[0, 0].bar(vendas_por_produto.index, vendas_por_produto.values, color='skyblue')
axes[0, 0].set_title('Volume Total de Vendas')
axes[0, 0].set_ylabel('Unidades Vendidas')
axes[0, 0].tick_params(axis='x', rotation=45)

# Revenue por produto
revenue_por_produto = df_vendas.groupby('product_name')['revenue'].sum().sort_values(ascending=False)
axes[0, 1].bar(revenue_por_produto.index, revenue_por_produto.values, color='lightgreen')
axes[0, 1].set_title('Revenue por Produto')
axes[0, 1].set_ylabel('Revenue (R$)')
axes[0, 1].tick_params(axis='x', rotation=45)

# Distribuição de preços por produto
df_vendas.boxplot(column='sale_price', by='product_name', ax=axes[1, 0])
axes[1, 0].set_title('Distribuição de Preços')
axes[1, 0].set_xlabel('Produto')
axes[1, 0].tick_params(axis='x', rotation=45)

# Quantidade por transação
df_vendas.boxplot(column='quantity', by='product_name', ax=axes[1, 1])
axes[1, 1].set_title('Distribuição de Quantidades')
axes[1, 1].set_xlabel('Produto')
axes[1, 1].tick_params(axis='x', rotation=45)

plt.tight_layout()
plt.show()
📅 Análise 2: Padrões Temporais e Sazonalidade 
print(f"\n📅 ANÁLISE 2: SAZONALIDADE E PADRÕES TEMPORAIS")
print("=" * 50)

# Vendas por mês
vendas_mensais = df_vendas.groupby(['month', 'product_name'])['quantity'].sum().reset_index()
vendas_mensais_pivot = vendas_mensais.pivot(index='month', columns='product_name', values='quantity')

print("📊 VENDAS MENSAIS POR PRODUTO:")
print(vendas_mensais_pivot)

# Calcular índice de sazonalidade
print(f"\n🎯 ÍNDICE DE SAZONALIDADE (média mensal / média geral):")
for produto in vendas_mensais_pivot.columns:
    media_geral = vendas_mensais_pivot[produto].mean()
    print(f"\n{produto}:")
    for mes, vendas in vendas_mensais_pivot[produto].items():
        indice = vendas / media_geral
        mes_nome = ['Jan', 'Fev', 'Mar', 'Abr', 'Mai', 'Jun'][mes-1]
        print(f"  {mes_nome}: {indice:.2f} ", end="")
        if indice > 1.2:
            print("🔥 ALTA DEMANDA")
        elif indice < 0.8:
            print("❄️ BAIXA DEMANDA")
        else:
            print("⚪ NORMAL")

# Análise por dia da semana
vendas_por_dia = df_vendas.groupby('weekday')['quantity'].sum()
dias_semana = ['Segunda', 'Terça', 'Quarta', 'Quinta', 'Sexta', 'Sábado', 'Domingo']

print(f"\n📅 PADRÃO SEMANAL:")
for i, vendas in vendas_por_dia.items():
    print(f"  {dias_semana[i]}: {vendas} unidades")

# Visualização temporal
fig, axes = plt.subplots(2, 2, figsize=(16, 10))
fig.suptitle('TechStore - Análise Temporal', fontsize=16, fontweight='bold')

# Vendas por mês (heatmap)
sns.heatmap(vendas_mensais_pivot.T, annot=True, fmt='.0f', cmap='YlOrRd', ax=axes[0, 0])
axes[0, 0].set_title('Sazonalidade por Produto')
axes[0, 0].set_xlabel('Mês')

# Vendas diárias
vendas_diarias = df_vendas.groupby('sale_date')['quantity'].sum()
axes[0, 1].plot(vendas_diarias.index, vendas_diarias.values, color='blue', alpha=0.7)
axes[0, 1].set_title('Vendas Diárias')
axes[0, 1].set_ylabel('Unidades')

# Padrão semanal
axes[1, 0].bar([dias_semana[i] for i in vendas_por_dia.index], vendas_por_dia.values, color='orange')
axes[1, 0].set_title('Padrão Semanal')
axes[1, 0].set_ylabel('Unidades Vendidas')
axes[1, 0].tick_params(axis='x', rotation=45)

# Revenue semanal por canal
revenue_canal = df_vendas.groupby(['weekday', 'sales_channel'])['revenue'].sum().reset_index()
revenue_pivot = revenue_canal.pivot(index='weekday', columns='sales_channel', values='revenue')
revenue_pivot.plot(kind='bar', stacked=True, ax=axes[1, 1])
axes[1, 1].set_title('Revenue por Canal e Dia')
axes[1, 1].set_xlabel('Dia da Semana')
axes[1, 1].legend(title='Canal')

plt.tight_layout()
plt.show()

⚠️ Insights para Gestão de Estoque

Produtos com alta variabilidade precisam de estoque de segurança maior
Padrões sazonais permitem planejar compras antecipadamente
Dias da semana afetam demanda - ajustar reposição accordingly
Canais diferentes têm padrões distintos - estoque por canal

Correlações e Insights para IA

Vamos identificar correlações que podem orientar algoritmos de previsão de demanda e otimização de estoque.

🔗 Análise de Correlações para Estoque 
print("🔗 ANÁLISE 3: CORRELAÇÕES PARA ALGORITMOS DE IA")
print("=" * 50)

# Preparar dados para correlação
df_corr = df_vendas.copy()
df_corr['is_weekend'] = (df_corr['weekday'] >= 5).astype(int)
df_corr['is_premium'] = (df_corr['customer_segment'] == 'Premium').astype(int)
df_corr['is_online'] = (df_corr['sales_channel'] == 'Online').astype(int)

# Selecionar variáveis numéricas relevantes
numeric_vars = ['quantity', 'sale_price', 'revenue', 'month', 'weekday', 
               'is_weekend', 'is_premium', 'is_online']

correlation_matrix = df_corr[numeric_vars].corr()

print("📊 MATRIZ DE CORRELAÇÃO:")
print(correlation_matrix.round(3))

# Visualizar correlações
plt.figure(figsize=(12, 8))
mask = np.triu(np.ones_like(correlation_matrix, dtype=bool))
sns.heatmap(correlation_matrix, mask=mask, annot=True, cmap='RdBu_r', center=0,
            square=True, linewidths=0.5, fmt='.3f')
plt.title('Correlações - TechStore Dataset', fontsize=14, fontweight='bold')
plt.tight_layout()
plt.show()

# Identificar correlações importantes para estoque
print(f"\n🎯 CORRELAÇÕES RELEVANTES PARA IA:")

important_correlations = [
    ('quantity', 'is_weekend', "Fim de semana afeta demanda?"),
    ('quantity', 'month', "Há sazonalidade mensal?"),
    ('sale_price', 'quantity', "Preço afeta quantidade vendida?"),
    ('revenue', 'is_premium', "Cliente premium gera mais revenue?"),
    ('quantity', 'is_online', "Canal online vende mais?")
]

for var1, var2, question in important_correlations:
    corr_value = correlation_matrix.loc[var1, var2]
    print(f"\n❓ {question}")
    print(f"   📊 Correlação {var1} × {var2}: {corr_value:.3f}")
    
    if abs(corr_value) >= 0.3:
        strength = "forte" if abs(corr_value) >= 0.7 else "moderada"
        direction = "positiva" if corr_value > 0 else "negativa"
        print(f"   ✅ Correlação {strength} {direction} - ÚTIL para algoritmos de IA")
    else:
        print(f"   ⚪ Correlação fraca - variáveis independentes")

# Análise específica por produto
print(f"\n📱 ANÁLISE POR PRODUTO - SMARTPHONE vs NOTEBOOK:")

smartphone_data = df_vendas[df_vendas['product_name'] == 'Smartphone']
notebook_data = df_vendas[df_vendas['product_name'] == 'Notebook']

print(f"Smartphone - Correlação quantidade × mês: {smartphone_data['quantity'].corr(smartphone_data['month']):.3f}")
print(f"Notebook - Correlação quantidade × mês: {notebook_data['quantity'].corr(notebook_data['month']):.3f}")

# Insights para algoritmos de ML
print(f"\n🤖 INSIGHTS PARA ALGORITMOS DE IA:")
print(f"-" * 40)
print(f"📊 Features importantes para modelos de previsão:")
print(f"   • Mês (sazonalidade)")
print(f"   • Dia da semana (padrão semanal)")
print(f"   • Tipo de produto (comportamentos diferentes)")
print(f"   • Canal de venda (padrões distintos)")
print(f"   • Segmento de cliente (diferentes volumes)")
print(f"\n🎯 Algoritmos recomendados:")
print(f"   • Séries temporais (ARIMA, Prophet) para sazonalidade")

print(f"   • Clustering para segmentar produtos similares")
